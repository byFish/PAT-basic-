#include <stdio.h>
#include <math.h>


int main() {
	int n, m;
	int k = 0, h = 0;//ËØÊıÊı
	int iss;
	scanf("%d%d", &n, &m);
	if (n == 1) {
		printf("2");
		h++;
	}
	k++;
	for (int i = 3;; i++) {
		iss = 0;
		for (int j = 2; j <= sqrt(i); j++) {
			if (i%j == 0) {
				iss = 1;
			}
		}
		if (iss == 0) {			
			k++;
			if (k >= n) {
				if (h % 10 == 0 && h != 0)
					printf("\n");
				else if(h != 0)
					printf(" ");
				printf("%d", i);
				h++;
			}
			if (k >= m)
				break;
		}
	}
	return 0;
}
