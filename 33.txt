#include <stdio.h>
#include <string.h>
#include <math.h>


int main() {
	int bb[200] = {0};//ÀÃµôµÄ¼üÎª1
	char n[200], m[100005];
	for (int i = 0;; i++) {
		scanf("%c", &n[i]);
		if (n[i] == '\n') {
			n[i] = '\0';
			break;
		}
		bb[n[i]] = 1;
		if (n[i] >= 'a' && n[i] <= 'z') {
			n[i] -= ('a' - 'A');
			bb[n[i]] = 1;
		}
		else if (n[i] >= 'A' && n[i] <= 'Z') {
			n[i] += ('a' - 'A');
			bb[n[i]] = 1;
		}
	}
	
	for (int i = 0;; i++) {
		scanf("%c", &m[i]);
		if (m[i] == '\n') {
			m[i] = '\0';
			break;
		}
		if (bb[m[i]] != 0)
			continue;
		if (bb['+'] == 1 && m[i] >= 'A' && m[i] <= 'Z')
			continue;
		printf("%c", m[i]);
	}
	printf("\n");
	return 0;
}
